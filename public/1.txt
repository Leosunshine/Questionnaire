public function testencAction()
{
    $key = pack('H*', "bcb04b7e103a0cd8b54763051cef08bc55abe029fdebae5e1d417e2ffb2a00a3");
    $text = "孙大圣";
    $iv = pack('H*', "bcb04b7e103a0cd8b54763051cef08bc55abe029fdebae5e1d417e2ffb2a00a3");
    $cilper = unpack('H*', mcrypt_encrypt(MCRYPT_RIJNDAEL_256, $key, $text, MCRYPT_MODE_CBC, $iv));
    $str = '';
    foreach ($cilper as $c)
    {
        $str = $str.$c;
    }
    echo $str;
}

public function  testdecAction($cilper)
{
    $key = pack('H*', "bcb04b7e103a0cd8b54763051cef08bc55abe029fdebae5e1d417e2ffb2a00a3");
    $iv = pack('H*', "bcb04b7e103a0cd8b54763051cef08bc55abe029fdebae5e1d417e2ffb2a00a3");
    echo mcrypt_decrypt(MCRYPT_RIJNDAEL_256, $key, pack('H*', $cilper), MCRYPT_MODE_CBC, $iv);
}
